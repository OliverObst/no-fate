"alt" $alt "caption" ($.Get $captionKey)) -}} {{- end -}} {{- end -}} {{- if not $images -}} {{- errorf "No Fate image-spread: at least src and alt are required at %s" .Position -}} {{- end -}} {{- $layout := lower (.Get "layout" | default (cond (eq (len $images) 1) "single-wide" (cond (eq (len $images) 2) "pair" "triptych"))) -}} {{- if not (in (slice "single-wide" "pair" "triptych" "asymmetric") $layout) -}} {{- errorf "No Fate image-spread: unsupported layout %q at %s" $layout .Position -}} {{- end -}} {{- if and (eq $layout "single-wide") (ne (len $images) 1) -}} {{- errorf "No Fate image-spread: single-wide accepts exactly one image at %s" .Position -}} {{- end -}} {{- if and (eq $layout "pair") (ne (len $images) 2) -}} {{- errorf "No Fate image-spread: pair accepts exactly two images at %s" .Position -}} {{- end -}} {{- if and (eq $layout "triptych") (ne (len $images) 3) -}} {{- errorf "No Fate image-spread: triptych accepts exactly three images at %s" .Position -}} {{- end -}}
